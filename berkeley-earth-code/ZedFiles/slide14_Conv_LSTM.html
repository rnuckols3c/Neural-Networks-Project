import pandas as pd
import numpy as np
import torch
from sklearn.preprocessing import MinMaxScaler
from torch.utils.data import Dataset, DataLoader

# === Load Dataset ===
df = pd.read_csv("/Users/rero/.cache/kagglehub/datasets/berkeleyearth/climate-change-earth-surface-temperature-data/versions/2/GlobalLandTemperaturesByMajorCity.csv")
df = df.dropna(subset=["AverageTemperature"])
df["dt"] = pd.to_datetime(df["dt"])

# === Filter for Canada and modern data ===
df = df[df["Country"] == "Canada"]
df = df[df["dt"].dt.year >= 1950]

# === Convert Latitude/Longitude ===
def parse_coord(val):
    if val[-1] in ["N", "E"]:
        return float(val[:-1])
    else:
        return -float(val[:-1])

df["Latitude"] = df["Latitude"].apply(parse_coord)
df["Longitude"] = df["Longitude"].apply(parse_coord)

# === Bin spatial dimensions ===
df["lat_bin"] = pd.cut(df["Latitude"], bins=np.arange(40, 90, 2))
df["lon_bin"] = pd.cut(df["Longitude"], bins=np.arange(-140, -50, 2))

# === Monthly aggregation ===
df["month"] = df["dt"].dt.to_period("M")
heatmaps = df.groupby(["month", "lat_bin", "lon_bin"], observed=False)["AverageTemperature"].mean().reset_index()

# === Pivot to cube: (time, lat_bin, lon_bin) ===
cube = heatmaps.pivot_table(index="month", columns=["lat_bin", "lon_bin"], values="AverageTemperature", observed=False)
cube = cube.ffill().bfill().fillna(0)

# === Normalize + reshape to (time, 1, H, W) ===
scaler = MinMaxScaler()
cube_norm = scaler.fit_transform(cube)
cube_tensor = torch.tensor(cube_norm, dtype=torch.float32)
cube_tensor = cube_tensor.view(-1, 1, *cube.shape[1:])

# === Sliding window sequences ===
def make_sequences(data, input_len=12, pred_len=1):
    X, y = [], []
    for i in range(len(data) - input_len - pred_len):
        X.append(data[i:i+input_len])
        y.append(data[i+input_len:i+input_len+pred_len])
    return torch.stack(X), torch.stack(y)

X_seq, y_seq = make_sequences(cube_tensor)

# === Dataset class ===
class GridSequenceDataset(Dataset):
    def __init__(self, X, y):
        self.X = X
        self.y = y

    def __len__(self):
        return len(self.X)

    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]

# === Train/test split ===
split = int(0.8 * len(X_seq))
train_set = GridSequenceDataset(X_seq[:split], y_seq[:split])
test_set = GridSequenceDataset(X_seq[split:], y_seq[split:])
train_loader = DataLoader(train_set, batch_size=4, shuffle=True)
test_loader = DataLoader(test_set, batch_size=4)

import torch
import torch.nn as nn

class ConvLSTMCell(nn.Module):
    def __init__(self, input_channels, hidden_channels, kernel_size):
        super().__init__()
        padding = kernel_size // 2
        self.conv = nn.Conv2d(
            in_channels=input_channels + hidden_channels,
            out_channels=4 * hidden_channels,
            kernel_size=kernel_size,
            padding=padding
        )
        self.hidden_channels = hidden_channels

    def forward(self, input_tensor, h_cur, c_cur):
        combined = torch.cat([input_tensor, h_cur], dim=1)  # (B, C+H, H, W)
        conv_output = self.conv(combined)
        cc_i, cc_f, cc_o, cc_g = torch.chunk(conv_output, 4, dim=1)

        i = torch.sigmoid(cc_i)
        f = torch.sigmoid(cc_f)
        o = torch.sigmoid(cc_o)
        g = torch.tanh(cc_g)

        c_next = f * c_cur + i * g
        h_next = o * torch.tanh(c_next)
        return h_next, c_next

class ConvLSTM(nn.Module):

        self.num_layers = num_layers
        self.hidden_channels = hidden_channels

import torch
import torch.nn as nn
import torch.optim as optim
import matplotlib.pyplot as plt

# === Device ===
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# === ConvLSTM Model Definition ===
class ConvLSTMCell(nn.Module):
    def __init__(self, input_channels, hidden_channels, kernel_size):
        super().__init__()
        padding = kernel_size // 2
        self.conv = nn.Conv2d(
            in_channels=input_channels + hidden_channels,
            out_channels=4 * hidden_channels,
            kernel_size=kernel_size,
            padding=padding
        )
        self.hidden_channels = hidden_channels

    def forward(self, input_tensor, h_cur, c_cur):
        combined = torch.cat([input_tensor, h_cur], dim=1)
        conv_output = self.conv(combined)
        cc_i, cc_f, cc_o, cc_g = torch.chunk(conv_output, 4, dim=1)
        i = torch.sigmoid(cc_i)
        f = torch.sigmoid(cc_f)
        o = torch.sigmoid(cc_o)
        g = torch.tanh(cc_g)
        c_next = f * c_cur + i * g
        h_next = o * torch.tanh(c_next)
        return h_next, c_next

class ConvLSTM(nn.Module):
    def __init__(self, input_channels, hidden_channels, kernel_size, num_layers):
        super().__init__()
        self.num_layers = num_layers
        self.hidden_channels = hidden_channels
        self.layers = nn.ModuleList()

        for i in range(num_layers):
            in_ch = input_channels if i == 0 else hidden_channels
            self.layers.append(ConvLSTMCell(in_ch, hidden_channels, kernel_size))

        self.conv_out = nn.Conv2d(hidden_channels, 1, kernel_size=1)

    def forward(self, x):  # x: (B, T, C, H, W)
        B, T, C, H, W = x.size()
        h, c = [None] * self.num_layers, [None] * self.num_layers

        for layer_idx in range(self.num_layers):
            h[layer_idx] = torch.zeros(B, self.hidden_channels, H, W, device=x.device)
            c[layer_idx] = torch.zeros(B, self.hidden_channels, H, W, device=x.device)

        for t in range(T):
            input_t = x[:, t]
            for layer_idx, layer in enumerate(self.layers):
                h[layer_idx], c[layer_idx] = layer(input_t, h[layer_idx], c[layer_idx])
                input_t = h[layer_idx]

        return self.conv_out(h[-1])  # Output: (B, 1, H, W)

# === Instantiate model ===
convlstm_model = ConvLSTM(input_channels=1, hidden_channels=16, kernel_size=3, num_layers=2).to(device)
criterion = nn.MSELoss()
optimizer = optim.Adam(convlstm_model.parameters(), lr=0.001)

# === Training Loop ===
num_epochs = 50
train_losses, val_losses = [], []

for epoch in range(num_epochs):
    convlstm_model.train()
    train_loss = 0.0
    for inputs, targets in train_loader:
        inputs, targets = inputs.to(device), targets.to(device)

        outputs = convlstm_model(inputs.unsqueeze(2))  # (B, T, C=1, H, W)
        target_resized = targets.squeeze(1).unsqueeze(1)  # (B, 1, W)  (B, 1, 1, W)
        loss = criterion(outputs, target_resized)

        loss.backward()
        optimizer.step()
        optimizer.zero_grad()
        train_loss += loss.item()

    convlstm_model.eval()
    val_loss = 0.0
    with torch.no_grad():
        for inputs, targets in test_loader:
            inputs, targets = inputs.to(device), targets.to(device)

            outputs = convlstm_model(inputs.unsqueeze(2))
            target_resized = targets.squeeze(1).unsqueeze(1)
            loss = criterion(outputs, target_resized)
            val_loss += loss.item()

    train_losses.append(train_loss / len(train_loader))
    val_losses.append(val_loss / len(test_loader))
    print(f"[ConvLSTM] Epoch {epoch+1}/{num_epochs}  Train Loss: {train_losses[-1]:.4f}  Val Loss: {val_losses[-1]:.4f}")

# === Loss Curve Plot ===
plt.figure(figsize=(10, 4))
plt.plot(train_losses, label="Train Loss", color="blue")
plt.plot(val_losses, label="Val Loss", color="orange")
plt.title("Conv-LSTM: Training vs Validation Loss")
plt.xlabel("Epoch")
plt.ylabel("MSE Loss")
plt.legend()
plt.grid(True, linestyle="--", alpha=0.5)

plt.show()
